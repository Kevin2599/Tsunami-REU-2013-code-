% 1.1  Define tsunami wave (versus wind wave)
	\slide[Tsunami Waves] {
	\begin{enumerate}
		\item Tsunami waves are long waves, $\lambda>> D$ or $\frac{D}{\lambda}\approx0$.
		
		\item Wind waves are short waves, $\lambda\not>\not> D$ or $\frac{D}{\lambda}\not\approx 0$.
		\end{enumerate}
